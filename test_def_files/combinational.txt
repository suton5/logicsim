START DEVICES;
SW1 = SWITCH, init=0;
SW2 = SWITCH, init=1;
SIGGEN1 = SIGGEN, sig=00100;
NOR1 = NOR, ip=2;
NAND1 = NAND, ip=2;
OR1 = OR, ip=2;
END DEVICES;

START CONNECTIONS;
SW1 -> NOR1.I1;
SIGGEN1 -> NOR1.I2, NAND1.I1;
SW2 -> NAND1.I2 ;
NOR1 -> OR1.I1;
NAND1 -> OR1.I2;
END CONNECTIONS;

START MONITORS;
OR1;
END MONITORS;
