START DEVICES;
SW1 = SWITCH, init=0;
CLK1 = CLOCK, cycles=20;
G1 = NAND, ip=2;
G2 = NAND, ip=2;
G3 = NAND, ip=2;
G4 = NAND, ip=2;
END DEVICES;

START CONNECTIONS;
SW1 -> G1.I1;
CLK1 -> G1.I2, G2.I2;
G1 -> G2.I1, G3.I1 ;
G2 -> G4.I2;
G3 -> G4.I1;
G4 -> G3.I2;
END CONNECTIONS;

START MONITORS;
G2; 
G3;
END MONITORS;
